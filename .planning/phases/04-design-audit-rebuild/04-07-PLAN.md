**Add useEffect for audio-reactive bars:**
```typescript
useEffect(() => {
  const analyser = analyserRef.current
  if (!analyser) {
    // Idle animation — gentle random bars
    const interval = setInterval(() => {
      setBarHeights(Array.from({ length: 14 }, () => 15 + Math.random() * 30))
    }, 300)
    return () => clearInterval(interval)
  }

  const bufferLength = analyser.frequencyBinCount
  const dataArray = new Uint8Array(bufferLength)
  let rafId: number

  const update = () => {
    analyser.getByteFrequencyData(dataArray)
    const step = Math.floor(bufferLength / 14)
    const heights = Array.from({ length: 14 }, (_, i) => {
      const val = dataArray[i * step] || 0
      return Math.max(8, (val / 255) * 100)
    })
    setBarHeights(heights)
    rafId = requestAnimationFrame(update)
  }
  rafId = requestAnimationFrame(update)
  return () => cancelAnimationFrame(rafId)
}, [analyserRef])
```
